        // LOGICA DE ABAS ESTILIZADAS E STATUS DA ABA SOLICITAÇÕES
        function filtrarStatusOperacional(abaClicada, statusFiltro) {
            const abasContainer = abaClicada.closest('.sig-tabs');
            abasContainer.querySelectorAll('.sig-tab').forEach(aba => {
                aba.classList.remove('ativa');
                const badge = aba.querySelector('span');
                if(badge) {
                    badge.classList.remove('bg-blue-100', 'text-blue-800');
                    badge.classList.add('bg-slate-100', 'text-slate-500');
                }
            });
            abaClicada.classList.add('ativa');
            const novoBadge = abaClicada.querySelector('span');
            if(novoBadge) {
                novoBadge.classList.remove('bg-slate-100', 'text-slate-500');
                novoBadge.classList.add('bg-blue-100', 'text-blue-800');
            }

            const linhas = document.querySelectorAll('.sol-linha');
            linhas.forEach(linha => {
                const trStatus = linha.getAttribute('data-status');
                if(trStatus === statusFiltro) linha.style.display = 'table-row';
                else linha.style.display = 'none';
            });
        }

        function recalcularBadges() {
            const linhas = document.querySelectorAll('.sol-linha');
            let pendentes = 0, andamento = 0, concluidas = 0;
            linhas.forEach(linha => {
                const s = linha.getAttribute('data-status');
                if(s === 'PENDENTES') pendentes++;
                if(s === 'EM ANDAMENTO') andamento++;
                if(s === 'CONCLUIDAS') concluidas++;
            });

            const bPen = document.getElementById('badge-pendentes');
            if(bPen) bPen.innerText = pendentes;
            
            const bAnd = document.getElementById('badge-andamento');
            if(bAnd) bAnd.innerText = andamento;
            
            const bCon = document.getElementById('badge-concluidas');
            if(bCon) bCon.innerText = concluidas;
        }

        function getGoEngine() {
            try {
                let win = window;
                let depth = 0;
                while(win && depth < 5) {
                    if(win.go) return win.go;
                    if(win === win.parent) break;
                    win = win.parent;
                    depth++;
                }
            } catch(e) {
                return { error: "cross-origin ou outro: " + e.message };
            }
            return null;
        }

        async function carregarTabelaComprasPedido() {
            const tb = document.getElementById('tabela-solicitacoes');
            try {
                const goEngine = getGoEngine();
                if(!goEngine) {
                    tb.innerHTML = '<tr><td colspan="7">Não achou goEngine na hierarquia</td></tr>';
                    return;
                }
                if(goEngine.error) {
                    tb.innerHTML = `<tr><td colspan="7">${goEngine.error}</td></tr>`;
                    return;
                }
                if(!goEngine.main || !goEngine.main.App || !goEngine.main.App.ListarSolicitacoesCompra) {
                    tb.innerHTML = '<tr><td colspan="7">goEngine.main.App incompleto!</td></tr>';
                    return;
                }
                
                const compras = await goEngine.main.App.ListarSolicitacoesCompra();
                tb.innerHTML = '';
                if(!compras) {
                    tb.innerHTML = '<tr><td colspan="7">compras retornou NULL</td></tr>';
                    return;
                }
                if(compras.length === 0) {
                    recalcularBadges();
                    tb.innerHTML = '<tr><td colspan="7" class="text-center py-10 font-bold text-slate-400">NENHUMA SOLICITAÇÃO ENCONTRADA.</td></tr>';
                    return;
                }

                let h = '';
                for(let s of compras) {
                    let badgeCad = s.codigo_erp ? `<span class="sig-badge sig-badge-success !text-[10px]">CÓD: ${s.codigo_erp}</span>` : `<span class="sig-badge sig-badge-error !text-[10px]">NOVO (S/ CÓD)</span>`;
                    
                    let bgStatus = '';
                    let statusLabel = s.status || 'PENDENTES';
                    if (statusLabel === 'PENDENTES') bgStatus = 'sig-badge-warning';
                    else if (statusLabel === 'EM ANDAMENTO') bgStatus = 'sig-badge-blue';
                    else bgStatus = 'sig-badge-success';

                    const sData = {
                        id: s.codigo_erp || '',
                        descricao: s.produto,
                        descricao_tecnica: s.produto,
                        marca: s.marca,
                        referencia: s.referencia,
                        quantidade_solicitada: s.quantidade,
                        solicitacao_id: s.id
                    };
                    const sJson = JSON.stringify(sData).replace(/'/g, "&#39;").replace(/"/g, "&quot;");

                    let colAcoes = `
                        <td class="text-center">
                            <button class="bg-sig-blue text-white rounded-sm px-3 py-1 text-[9px] font-black hover:bg-blue-800 shadow-sm w-[130px]" onclick="selecionarProdutoParaPedido(${sJson})">
                                VINCULAR AO PEDIDO
                            </button>
                        </td>
                    `;

                    h += `
                        <tr class="hover:bg-blue-50 cursor-pointer border-t border-slate-200 transition-colors group sol-linha" data-status="${statusLabel}">
                            <td class="text-center font-mono text-[11px] text-slate-500 font-bold whitespace-nowrap">${s.data_solicitacao}</td>
                            <td class="font-bold text-[12px] text-slate-700">${s.solicitante_nome}</td>
                            <td class="uppercase tracking-tight text-slate-800">
                                <span class="font-bold text-[12px]">${s.produto} ${s.marca ? ' - '+s.marca : ''}</span>
                                <div class="text-[10px] text-slate-400 font-normal mt-0.5">${s.observacao}</div>
                            </td>
                            <td class="text-center font-mono font-bold text-blue-700 text-[12px]">${s.quantidade}</td>
                            <td class="text-center">${badgeCad}</td>
                            <td class="text-center">
                                <span class="sig-badge ${bgStatus} !text-[10px]">${statusLabel}</span>
                            </td>
                            ${colAcoes}
                        </tr>
                    `;
                }
                tb.innerHTML = h;
                recalcularBadges();
                
                // Mostrar a aba default "PENDENTES" explicitamente
                const tabPendentes = document.querySelector('#aba-solicitacoes .sig-tabs .sig-tab:first-child');
                if (tabPendentes) filtrarStatusOperacional(tabPendentes, 'PENDENTES');

            } catch(e) {
                console.error("Erro ao listar:", e);
            }
        }

            // Limite de altura melhorado: 
            // - Mínimo 180px para garantir a visualização da imagem quadrada 124x124 + legendas.
            if (newHeight < 180) newHeight = 180;
            if (newHeight > 600) newHeight = 600; // Aumentado para permitir visão maior das aplicações

            painelInferior.style.height = `${newHeight}px`;
        }

        window.addEventListener('keydown', (e) => {
            if (e.key === 'Escape') {
                const modalApp = document.getElementById('modal-nova-aplicacao');
                if (!modalApp.classList.contains('hidden')) {
                    fecharModalAplicacao();
                    return;
                }
                const modalMarca = document.getElementById('modal-selecao-marca');
                if (!modalMarca.classList.contains('hidden')) {
                    fecharSelecaoMarca();
                    return;
                }

                fecharJanelaSelecao();
                return;
            }
            if (e.key === 'F2') {
                e.preventDefault();
                mudarAbaPesquisa('catalogo');
            }
            if (e.key === 'F3') {
                const modalCadastro = document.getElementById('modal-cadastro');
                if (modalCadastro.classList.contains('hidden')) {
                    e.preventDefault();
                    mudarAbaPesquisa('solicitacoes');
                }
            }
            if (e.key === 'F6') {
                const abaCat = document.getElementById('aba-f2');
                if (!abaCat.classList.contains('hidden')) {
                    e.preventDefault();
                    if (window.abrirCadastroProduto) window.abrirCadastroProduto();
                }
            }
            if (e.key === 'F8') {
                e.preventDefault();
                pesquisarComFiltros();
            }
            if (e.key === 'F5') {
                if(window.isProdutoReadOnlyGlobal) return;
                const modalCadastro = document.getElementById('modal-cadastro');
                if (modalCadastro && !modalCadastro.classList.contains('hidden')) {
                    e.preventDefault();
                    if (window.salvarProduto) window.salvarProduto();
                }
            }
            
            // Navegação de Setas na Tabela
            if (e.key === 'ArrowDown' || e.key === 'ArrowUp') {
                const ativaTabelaOuNao = document.activeElement.tagName === 'INPUT' || document.activeElement.tagName === 'SELECT';
                if (ativaTabelaOuNao) return; 

                const tbody = document.getElementById('tb-produtos-lista');
                if (!tbody) return;
                const rows = Array.from(tbody.querySelectorAll('tr'));
                if (rows.length === 0) return;

                e.preventDefault();
                let idx = rows.findIndex(r => r.classList.contains('selected'));
                
                if (e.key === 'ArrowDown') {
                    if (idx < rows.length - 1) idx++;
                    else idx = 0; 
                } else if (e.key === 'ArrowUp') {
                    if (idx > 0) idx--;
                    else idx = rows.length - 1; 
                }

                if(idx !== -1) {
                    // Remove current selection
                    if (rows.find(r => r.classList.contains('selected'))) {
                        rows.forEach(r => r.classList.remove('selected', 'bg-blue-50'));
                    }
                    const rowSelecionada = rows[idx];
                    rowSelecionada.classList.add('selected', 'bg-blue-50');
                    rowSelecionada.scrollIntoView({ behavior: 'smooth', block: 'center' });
                    // No click unless the user presses Enter (because click selects the product for the sale)
                }
            }

            if (e.key === 'Enter') {
                const ativaTabelaOuNao = document.activeElement.tagName === 'INPUT' || document.activeElement.tagName === 'SELECT';
                if (!ativaTabelaOuNao) {
                    const selected = document.querySelector('#tb-produtos-lista tr.selected');
                    if (selected) {
                        e.preventDefault();
                        if (selected.onclick) {
                            selected.onclick(); // Seleciona pro pedido
                        }
                    }
                }
            }
        });

        function handleMouseMove(e) {
            if (!isResizing) return;

            const windowHeight = window.innerHeight;
            // O splitter está acima do painel, então a altura é o que falta da tela
            // subtraindo a barra de status (aprox 46px) e a margem de segurança
            const statusBarHeight = 46;

            // Limite de altura melhorado: 
            // - Mínimo 180px para garantir a visualização da imagem quadrada 124x124 + legendas.

                }

                confirmarSelecao();

        async function pesquisarComFiltros() {
            const engine = window.goEngine || window.go;
            if (!engine?.main?.App?.PesquisarProdutosAvancado) return;

            // Coleta filtros
            const f = {
                descricao: document.getElementById('filtro-descricao').value,
                cod_fabricante: document.getElementById('filtro-sku').value,
                marca_veiculo: document.getElementById('filtro-marca-veiculo').value,
                modelo_veiculo: document.getElementById('filtro-modelo-veiculo').value,
                versao_veiculo: document.getElementById('filtro-versao-veiculo').value,
                marca_peca: document.getElementById('filtro-marca-peca').value,

                localizacao: document.getElementById('filtro-localizacao').value,
                cod_barras: document.getElementById('filtro-ean').value,
                similar_de: document.getElementById('filtro-similares').value,
                filtrar_saldo: document.getElementById('filtro-saldo').value
            };

            // Se todos estiverem vazios (exceto saldo), carrega lista normal para evitar overhead
            const hasValue = Object.keys(f).some(k => k !== 'filtrar_saldo' && f[k].trim() !== '');
            if(!hasValue && f.filtrar_saldo === 'TODOS') {
                return carregarProdutosCentral();
            }

            setLoading(true);
            try {
                const produtos = await engine.main.App.PesquisarProdutosAvancado(f);
                cacheCentral = produtos || [];
                renderCentralGrid(cacheCentral);
            } catch (err) {
                console.error("Erro na pesquisa: ", err);
            } finally {
                setLoading(false);
            }
        }

        // --- LOGICA DE AUTOCOMPLETE INLINE (PALAVRA POR PALAVRA) ---
        let palavrasUnicas = [];
        function handleAutocomplete(e) {
            const input = e.target;
            const hint = document.getElementById('filtro-descricao-hint');
            const textoInteiro = input.value.toUpperCase();
            
            hint.value = "";
            if (!textoInteiro || palavrasUnicas.length === 0) return;

            // Pega a última palavra que está sendo digitada
            const partes = textoInteiro.split(' ');
            const ultimaPalavra = partes[partes.length - 1];
            
            if (!ultimaPalavra || ultimaPalavra.length < 2) return;

            // Busca sugestão que comece com a última palavra
            const match = palavrasUnicas.find(p => p.startsWith(ultimaPalavra));
            if (match) {
                // Monta o hint apenas para a palavra atual
                const textoAnterior = partes.slice(0, -1).join(' ');
                hint.value = (textoAnterior ? textoAnterior + ' ' : '') + match;
            }
        }

        function confirmAutocomplete(e) {
            if (e.key === 'Tab' || e.key === 'Enter') {
                const hint = document.getElementById('filtro-descricao-hint');
                const input = document.getElementById('filtro-descricao');
                if (hint.value && input.value !== hint.value) {
                    e.preventDefault();
                    input.value = hint.value + ' '; // Adiciona espaço para a próxima palavra
                    hint.value = "";
                    debounceFilter();
                }
            }
        }
