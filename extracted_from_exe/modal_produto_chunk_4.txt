                filtrar_saldo: document.getElementById('filtro-saldo').value

                setLoading(false);

        // --- LOGICA DE AUTOCOMPLETE INLINE (PALAVRA POR PALAVRA) ---
        let palavrasUnicas = [];
        function handleAutocomplete(e) {
            const input = e.target;
            const hint = document.getElementById('filtro-descricao-hint');
            const textoInteiro = input.value.toUpperCase();
            
            hint.value = "";
            if (!textoInteiro || palavrasUnicas.length === 0) return;

            // Pega a última palavra que está sendo digitada
            const partes = textoInteiro.split(' ');
            const ultimaPalavra = partes[partes.length - 1];
            
            if (!ultimaPalavra || ultimaPalavra.length < 2) return;

            // Busca sugestão que comece com a última palavra
            const match = palavrasUnicas.find(p => p.startsWith(ultimaPalavra));
            if (match) {
                // Monta o hint apenas para a palavra atual
                const textoAnterior = partes.slice(0, -1).join(' ');
                hint.value = (textoAnterior ? textoAnterior + ' ' : '') + match;
            }
        }

        function confirmAutocomplete(e) {
            if (e.key === 'Tab' || e.key === 'Enter') {
                const hint = document.getElementById('filtro-descricao-hint');
                const input = document.getElementById('filtro-descricao');
                if (hint.value && input.value !== hint.value) {
                    e.preventDefault();
                    input.value = hint.value + ' '; // Adiciona espaço para a próxima palavra
                    hint.value = "";
                    debounceFilter();
                }
            }
        }

                `;

                if(tfTotal) tfTotal.innerText = `${p.estoque_atual || '0'} ${p.unidade_sigla || 'UN'}`;
            }

            // 2. Imagem local via GO (Base64)
            const imgPreview = document.getElementById('img-preview-f2');
            if(imgPreview) {
                imgPreview.src = "";
                imgPreview.classList.add('hidden');
                
                if (engine && engine.main && engine.main.App) {
                    const b64 = await engine.main.App.ObterPrimeiraImagemB64(p.id);
                    if (b64) {
                        imgPreview.src = b64;
                        imgPreview.classList.remove('hidden');
                    }
                }
            }

            // Se engine carregado, busca sublistas (Lazy Loading) do Banco
            if (!engine?.main?.App) return;

            // 3. Aplicação
            const tbApp = document.getElementById('tb-aplicacao-lista');
            if(tbApp) {
                tbApp.innerHTML = '<tr><td colspan="5" class="text-center text-slate-400">Carregando...</td></tr>';
                try {
                    const apps = await engine.main.App.ListarAplicacoesDoProduto(p.id) || [];
                    tbApp.innerHTML = '';
                    if(apps.length === 0) tbApp.innerHTML = '<tr><td colspan="5" class="text-center text-slate-400 italic">Nenhuma aplicação cadastrada</td></tr>';
                    apps.forEach(a => {
                        tbApp.innerHTML += `
                            <tr>
                                <td>${a.marca || '-'}</td>
                                <td>${a.modelo || '-'}</td>
                                <td>${a.motor || '-'}</td>
                                <td>${a.ano_inicio || '-'}..${a.ano_fim || ''}</td>
                                <td>-</td>
                            </tr>
                        `;
                    });
                } catch(e) { tbApp.innerHTML = ''; }
            }

            // 4. Similares
            const tbSim = document.getElementById('tb-similares-lista');
            if(tbSim) {
                tbSim.innerHTML = '<tr><td colspan="3" class="text-center text-slate-400">Carregando...</td></tr>';
                try {
                    const convs = await engine.main.App.ListarConversoesDoProduto(p.id) || [];
                    tbSim.innerHTML = '';
                    if(convs.length === 0) tbSim.innerHTML = '<tr><td colspan="3" class="text-center text-slate-400 italic">Sem Similares vinculados</td></tr>';
                    convs.forEach(c => {
                        tbSim.innerHTML += `
                            <tr>
                                <td>${c.marca || '-'}</td>
                                <td>${c.codigo || '-'}</td>
                                <td class="text-right font-mono font-bold text-slate-400">-</td>
                            </tr>
                        `;
                    });
                } catch(e) { tbSim.innerHTML = ''; }
            }
        }

            try {
                const goEngine = window.parent.go || window.go;
                if (goEngine) {
                     // 1. Injeta Permissoes do HUB no Contexto Local
                     const permString = await goEngine.main.App.GetPermissoesLogado();
                     const permissoesArray = JSON.parse(permString || "[]");
                     window.SIG_AUTH = { modulos: {}, permissoes: {} };
                     permissoesArray.forEach(p => {
                         if(p.startsWith('MOD_')) window.SIG_AUTH.modulos[p] = true;
                         else window.SIG_AUTH.permissoes[p] = true;
                     });
                     window.SIG_AUTH.modulos['MOD_PRODUTO'] = true; // Permissão Local Mínima Garantida
                }
                
                // 2. Aplica as novas restrições Visuais imediatamente
                aplicarPermissoesNaTela();

                // 3. Processos Operacionais da Tela

                carregarProdutosCentral();

                aplicarPermissoesNaTela(); // Fallback mode
            }

        function renderListaUsuarios(lista) {
            const divLista = document.getElementById('vinc-lista-usuarios');

        function editarGrupo(g) {
            document.getElementById('grp-id').value = g.id;
            document.getElementById('grp-nome').value = g.nome;
            document.getElementById('grp-desc').value = g.descricao;
            document.getElementById('grp-ativo').checked = g.ativo;

            // Transforma o JSON do postgres de volta
            let obj = [];
            try {
                if(g.permissoes) obj = JSON.parse(g.permissoes);
            } catch(e){}

            const chks = document.querySelectorAll('.grp-perm');
            chks.forEach(c => {
                c.checked = obj.includes(c.value);

        async function salvarModal() {
            const nome = document.getElementById('grp-nome').value.toUpperCase().trim();
            if(!nome) {
                Toast.warning('O nome do Grupo é obrigatório!');
                return;
            }

            // Coleta a matriz de checkboxes (Array Flat)
            const permitidos = [];
            document.querySelectorAll('.grp-perm:checked').forEach(c => permitidos.push(c.value));

            const grupo = {
                id: parseInt(document.getElementById('grp-id').value),
                nome: nome,
                descricao: document.getElementById('grp-desc').value,
                permissoes: JSON.stringify(permitidos), // Stringifica array para o Postgres JSONB
                ativo: document.getElementById('grp-ativo').checked
            };

            const goEngine = window.parent.go || window.go;
            if(goEngine) {
                let resp = await goEngine.main.App.SalvarGrupoAcesso(grupo);
                if(resp === "OK") {
                    Toast.success('Grupo cadastrado e replicado no Banco.');
                    fecharModal();
                    carregarGrupos();
                } else {
                    Toast.error('Falha interna: ' + resp);
                }
            }
        }
