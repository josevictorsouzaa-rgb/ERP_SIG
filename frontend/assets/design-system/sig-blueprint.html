Como criar uma Tela Nova (DNA do Sistema)

// Use este Boilerplate para garantir que a tela 'orne' com o restante do SIG

<!-- DNA SIG: MANTENHA ESTA ESTRUTURA PARA TODA NOVA TELA -->
<!DOCTYPE html>
<html>
<head>
    <!-- 1. SEMPRE Importe o sig-ui.css -->
    <link rel="stylesheet" href="../assets/css/sig-ui.css">
</head>
<body>
    <!-- 2. Header Branding -->
    <header class="sig-hub-header">
        <div class="flex items-center gap-3">
            <span class="sig-logo-brand">SIG</span>
            <span class="sig-logo-separator">|</span>
            <span class="sig-logo-tagline">NOME DO MÓDULO</span>
        </div>
    </header>

    <div class="sig-app-container">
        <!-- 3. Sidebar Padrão (Conforme Blueprint) -->
        <aside class="sig-sidebar">
            <div class="sig-sidebar-header">
                <span>NAVEGAÇÃO</span>
            </div>
            <nav class="sig-sidebar-menu">
                <div class="sig-menu-item ativa">
                    <span>Item Principal</span>
                    <span class="sig-shortcut">F2</span>
                </div>
            </nav>
        </aside>

        <!-- 4. Área de Conteúdo Sóbria -->
        <main class="sig-main-content">
             <!-- Seu conteúdo aqui -->
        </main>
    </div>
</body>
</html>
