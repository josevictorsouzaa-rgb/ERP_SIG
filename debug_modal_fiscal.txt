        function fecharJanela() {
            if (window.runtime) window.runtime.Quit();
        }

        function mostrarErroReversivel(boxId, textId, msg) {
            const errorBox = document.getElementById(boxId);
            const errorText = document.getElementById(textId);
            errorText.innerText = msg;
            errorBox.style.display = 'flex';
            setTimeout(() => {
                errorBox.style.display = 'none';
            }, 5000);
        }

        async function handleLogin(e) {
            e.preventDefault();
            const btn = e.target.querySelector('button');
            const originalText = btn.innerHTML;
            btn.innerHTML = '<span class="animate-pulse">AUTENTICANDO...</span>';
            btn.disabled = true;

            const user = document.getElementById('username').value.trim();
            const pass = document.getElementById('password').value.trim();

            try {
                const goEngine = window.parent.go || window.go;
                if (!goEngine) {
                    mostrarErroReversivel('login-error-box', 'login-error-text', 'Acesso Negado: App offline.');
                    btn.innerHTML = originalText;
                    btn.disabled = false;
                    return;
                }

                // Dispara pro BACKEND do GO as credenciais
                const res = await goEngine.main.App.AutenticarLogin(user, pass);
                if (res.status === "ok") {
                    // VERIFICAR PRIMEIRO ACESSO
                    if (res.usuario.precisa_alterar_senha) {
                        window._tempLoggedUser = res.usuario;
                        document.getElementById('login-form-box').style.display = 'none';
                        document.getElementById('reset-form-box').style.display = 'block';
                        btn.innerHTML = originalText;
                        btn.disabled = false;
                        return; // Para o fluxo aqui!
                    }

                    btn.innerHTML = '<span class="text-green-300">ACESSO AUTORIZADO</span>';
                    
                    // Fade out pro Hub (Crossfade)
                    document.body.style.transition = 'opacity 0.25s ease-in-out';
                    document.body.style.opacity = '0';
                    
                    setTimeout(() => {
                        if (window.parent) {
                            window.parent.postMessage({ type: 'LOGIN_SUCCESS', usuario: res.usuario }, '*');
                        }
                    }, 250);
                } else {
                    mostrarErroReversivel('login-error-box', 'login-error-text', res.mensagem);
                    btn.innerHTML = originalText;
                    btn.disabled = false;
                }
            } catch (err) {
                mostrarErroReversivel('login-error-box', 'login-error-text', "Falha de Comunicação: " + err);
                btn.innerHTML = originalText;
                btn.disabled = false;
            }
        }

        async function handleResetSenha(e) {
            e.preventDefault();
            const pass1 = document.getElementById('new-password').value;
            const pass2 = document.getElementById('new-password-confirm').value;

            if (pass1 !== pass2) {
                mostrarErroReversivel('reset-error-box', 'reset-error-text', "ERRO: As novas credenciais não coincidem!");
                return;
            }
            if (pass1.length < 6) {
                mostrarErroReversivel('reset-error-box', 'reset-error-text', "ERRO: A credencial deve ter no mínimo 6 caracteres!");
                return;
            }

            const btn = e.target.querySelector('button');
            const originalText = btn.innerHTML;
            btn.innerHTML = '<span class="animate-pulse">ATUALIZANDO...</span>';
            btn.disabled = true;

            try {
                const goEngine = window.parent.go || window.go;
                const res = await goEngine.main.App.AlterarSenhaOperadorLogado(pass1);

                if (res === "OK") {
                    btn.innerHTML = '<span style="color: white;">CONCLUÍDO! INICIANDO...</span>';

                    // Como a senha foi atualizada no banco e na memória, podemos prosseguir pro Hub!
                    if (window._tempLoggedUser) {
                        window._tempLoggedUser.precisa_alterar_senha = false;
                    }
                    setTimeout(() => {
                        if (window.parent) {
                            window.parent.postMessage({ type: 'LOGIN_SUCCESS', usuario: window._tempLoggedUser }, '*');
                        }
                    }, 800);
                } else {
                    mostrarErroReversivel('reset-error-box', 'reset-error-text', "Erro ao redefinir a credencial: " + res);
                    btn.innerHTML = originalText;
                    btn.disabled = false;
                }
            } catch (err) {
                mostrarErroReversivel('reset-error-box', 'reset-error-text', "Falha Global: " + err);
                btn.innerHTML = originalText;
                btn.disabled = false;
            }
        }

        function togglePassword(inputId, iconId) {
            const input = document.getElementById(inputId);
            const icon = document.getElementById(iconId);
            if (input.type === 'password') {
                input.type = 'text';
                icon.innerText = 'visibility_off';
            } else {
                input.type = 'password';
                icon.innerText = 'visibility';
            }
        }

        function analisarForca(v1) {
            const str1 = document.getElementById('str-1');
            const str2 = document.getElementById('str-2');
            const str3 = document.getElementById('str-3');
            const txt = document.getElementById('str-text');

            if (v1.length === 0) {
                str1.style.background = '#e2e8f0';
                str2.style.background = '#e2e8f0';
                str3.style.background = '#e2e8f0';
                txt.innerText = '';
                return;
            }

            if (v1.length < 6) {
                str1.style.background = '#ef4444'; // Vermelho
                str2.style.background = '#e2e8f0';
                str3.style.background = '#e2e8f0';
                txt.innerText = 'FRACA';
                txt.style.color = '#ef4444';
            } else {
                // Tem 6 ou mais caracteres
                const temLetra = /[a-zA-Z]/.test(v1);
                const temNumero = /[0-9]/.test(v1);
                const temEspecial = /[^a-zA-Z0-9]/.test(v1);

                if ((temLetra && !temNumero && !temEspecial) || (!temLetra && temNumero && !temEspecial)) {
                    // Só letras ou só números
                    str1.style.background = '#f59e0b'; // Amarelo
                    str2.style.background = '#f59e0b';
                    str3.style.background = '#e2e8f0';
                    txt.innerText = 'MÉDIA (ADICIONE NÚMEROS E LETRAS PARA + SEGURANÇA)';
                    txt.style.color = '#f59e0b';
                } else {
                    // Mista = Forte
                    str1.style.background = '#10b981'; // Verde
                    str2.style.background = '#10b981';
                    str3.style.background = '#10b981';
                    txt.innerText = 'FORTE (ÓTIMA SELEÇÃO)';
                    txt.style.color = '#10b981';
                }
            }
        }

        function validarSenhas() {
            const v1 = document.getElementById('new-password').value;
            const v2 = document.getElementById('new-password-confirm').value;
            const btn = document.getElementById('btn-reset');
            const feedback = document.getElementById('pass-feedback');

            analisarForca(v1);

            if (v1.length === 0 && v2.length === 0) {
                feedback.innerText = '';
                desativarBtn(btn);
                return;
            }
            if (v1.length > 0 && v1.length < 6) {
                feedback.innerText = '';
                desativarBtn(btn);
                return;
            }
            if (v2.length > 0 && v1 !== v2) {
                feedback.innerText = 'NÃO CONFERE';
                feedback.style.color = '#ef4444';
                desativarBtn(btn);
                return;
            }
            if (v1 === v2 && v1.length >= 6) {
                feedback.innerText = 'VÁLIDA ✓';
                feedback.style.color = '#10b981';
                ativarBtn(btn);
            }
        }

        function desativarBtn(btn) {
            btn.disabled = true;
            btn.style.background = '#94a3b8';
            btn.style.boxShadow = 'none';
            btn.style.cursor = 'not-allowed';
        }
        function ativarBtn(btn) {
            btn.disabled = false;
            btn.style.background = '#10b981';
            btn.style.boxShadow = '0 10px 20px rgba(16,185,129,0.2)';
            btn.style.cursor = 'pointer';
        }
        function cancelarReset() {
            document.getElementById('reset-form-box').style.display = 'none';
            document.getElementById('login-form-box').style.display = 'block';
            document.getElementById('new-password').value = '';
            document.getElementById('new-password-confirm').value = '';
            validarSenhas();
        }
